    \node[draw=none] at ($ (x4)!0.5!(x5) + (0,0.6) $) {$\mathcal O$};
    \node[draw=none] at ($ (x6)!0.5!(x9) + (0,0.6) $) {$\mathcal E$};

    \draw (x7) -- (y4) [red, very thick];
    \draw (x8) -- (y5) [red, very thick];

    \node[draw=none] at (0,2) {\large $X$};
    \node[draw=none] at (0,0) {\large $Y$};
